Tailwind CSS Plus Element Templates

Factory functions for creating styled HTML elements. These are NOT Web Components — they create standard elements with Tailwind CSS styling.

Buttons

Solid Variants

Outline & Plain

Properties

PropTypeDefault
colorstring"dark/zinc"
outlinebooleanfalse
plainbooleanfalse
hrefstringnull
disabledbooleanfalse
 
import { createButton } from "../blocks/elements/actions/button.js";

const btn = createButton({ color: "blue" });
btn.textContent = "Click me";

// Outline button
const outline = createButton({ outline: true });

// Link styled as button
const link = createButton({ href: "/page" });

import { createFieldset, createField, createLabel } from "../blocks/elements/forms/fieldset.js";

const fieldset = createFieldset();
const field = createField();
field.appendChild(createLabel({ text: "Name" }));
field.appendChild(createInput());
fieldset.appendChild(field);
